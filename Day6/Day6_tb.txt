  logic         clk;
  logic         reset;
  logic         in;
  logic [3:0]   out;

  day6 DAY6 (.*);

    $dumpfile("day6.vcd");
    $dumpvars(1, day6_tb);
  end

    clk = 1'b1;

    clk = 1'b0;
    #5;

    end
